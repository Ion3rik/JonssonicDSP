22template <typename T>
23class CircularAudioBuffer {
24 public:
25 CircularAudioBuffer() = default;
26 ~CircularAudioBuffer() = default;
27
28 void resize(size_t newNumChannels, size_t newNumSamples) {
29 bufferSize = utils::nextPowerOfTwo(
30 newNumSamples); // ensure power-of-two size for efficient wrap-around
31 buffer.resize(newNumChannels, bufferSize);
32 writeIndex.assign(newNumChannels, 0);
33 }
34
35 // Write a sample to a channel (advances write index)
36 void write(size_t channel, T value) {
37 assert(channel < buffer.getNumChannels());
38 buffer[channel][writeIndex[channel]] = value;
39 writeIndex[channel] = (writeIndex[channel] + 1) & (bufferSize - 1);
40 }
41
42 // Read sample at delay (0 = most recent, 1 = previous, ...)
43 T read(size_t channel, size_t delay) const {
44 assert(channel < buffer.getNumChannels());
45 assert(delay < bufferSize);
46 size_t readIndex = (writeIndex[channel] + bufferSize - delay - 1) & (bufferSize - 1);
47 return buffer[channel][readIndex];
48 }
49
53 const T* readChannelPtr(size_t channel) const { return buffer.readChannelPtr(channel); }
54
58 T* writeChannelPtr(size_t channel) { return buffer.writeChannelPtr(channel); }
59
60 size_t getNumChannels() const { return buffer.getNumChannels(); }
61 size_t getBufferSize() const { return bufferSize; }
62 size_t getChannelWriteIndex(size_t channel) const { return writeIndex[channel]; }
63
64 void clear() {
65 buffer.clear(); // zero the buffer, does not deallocate
66 writeIndex.assign(buffer.getNumChannels(), 0); // reset indices
67 }
68
69 private:
70 AudioBuffer<T> buffer;
71 std::vector<size_t> writeIndex; // per-channel write index
72 size_t bufferSize = 0;
73};
