44 void prepare(size_t newNumChannels, T newSampleRate) {
45 numChannels = newNumChannels;
46 sampleRate = newSampleRate;
47 // Hardcoded cutoff frequency for DC blocking
48 constexpr T cutoffHz = T(10);
49 if (sampleRate > T(0)) {
50 R = T(1) - (T(2) * T(M_PI) * cutoffHz / sampleRate);

53 if (R > T(0.9999))

73 T processSample(size_t ch, T input) {
74 if (ch >= numChannels)
75 return input;
76 T y = input - x1[ch] + R * y1[ch];
77 x1[ch] = input;
78 y1[ch] = y;
