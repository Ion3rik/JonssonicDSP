53 void prepare(size_t newNumChannels, size_t newMaxBlockSize, T newSampleRate) {

57 eq.section(0).setResponse(BiquadFilter<T>::Response::Highpass); // LowCut

150 T computeVariableQ(T gainDb) {
151 if (gainDb < 0)
152 return BASE_Q * (T(1) + VARIABLE_Q_WEIGHT * std::abs(gainDb));
153 else
154 return BASE_Q / (T(1) + VARIABLE_Q_WEIGHT * gainDb);
