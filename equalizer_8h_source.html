58 void prepare(size_t newNumChannels, size_t newMaxBlockSize, T newSampleRate) {
59
60 // Store config variables

64 // Prepare biquad chain

66 eq.setType(0, BiquadType::Highpass); // LowCut

68 eq.setType(2, BiquadType::Peak); // HighMid

71 // Fixed parameters

163 T computeVariableQ(T gainDb) {
164 if (gainDb < 0)
165 return BASE_Q * (T(1) + VARIABLE_Q_WEIGHT * std::abs(gainDb));
166 else
167 return BASE_Q / (T(1) + VARIABLE_Q_WEIGHT * gainDb);
