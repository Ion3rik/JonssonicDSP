63 void prepare(size_t newNumChannels, T newSampleRate) {
64 // Store global parameters
65 numChannels = utils::detail::clampChannels(newNumChannels);
66 sampleRate = utils::detail::clampSampleRate(newSampleRate);
67
68 // Prepare DSP components
69 delayStage.prepare(numChannels, Time<T>::Milliseconds(MAX_DELAY_MS), sampleRate);
70 delayStage.setControlSmoothingTime(Time<T>::Milliseconds(T(SMOOTHING_TIME_MS)));
71 delayStage.setFeedforward(T(1), true); // 50/50 wet/dry mix internally for classic flanger sound
72 delayStage.setLfoType(Waveform::Triangle);
73
74 // Set Default Parameters
75 setRate(T(0.5), true); // 0.5 Hz
76 setDepth(T(0.5), true); // 50% depth
77 setFeedback(T(0.25), true); // 25% feedback
78 setDelayMs(T(2.0), true); // 2 ms center delay
79 setSpread(T(0.0), true); // No spread
80 }

85 void reset() { delayStage.reset(); }

96 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
97 delayStage.processBlock(input, output, numSamples);
98 utils::applyGain<T>(output, numChannels, numSamples, T(0.5));
99 }

105 void setRate(T rateHz, bool skipSmoothing = false) {

113 void setDepth(T depth, bool skipSmoothing = false) {

122 void setFeedback(T feedbackAmount, bool skipSmoothing = false) {
123 T scaledFeedback = std::clamp(feedbackAmount, T(-1.0), T(1.0)) * MAX_FEEDBACK;
124 delayStage.setFeedback(scaledFeedback, skipSmoothing);

134 }
135
142 void setSpread(T spread, bool skipSmoothing = false) {
143 // Update phase offsets based on spread (normalized to 0-1)
144 for (size_t ch = 0; ch < numChannels; ++ch) {
145 delayStage.setLfoPhaseOffset(ch, (spread * ch) / T(numChannels), skipSmoothing);
146 }
147 }
