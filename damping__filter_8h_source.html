38template <typename T>
39class DampingFilter<T, DampingType::OnePole> {
40 public:
42 DampingFilter() = default;
48 DampingFilter(size_t newNumChannels, T newSampleRate) { prepare(newNumChannels, newSampleRate); }
50 ~DampingFilter() = default;
51
57 void prepare(size_t newNumChannels, T newSampleRate) {
58 // Store config variables
59 numChannels = utils::detail::clampChannels(newNumChannels);
60 sampleRate = utils::detail::clampSampleRate(newSampleRate);
61
62 // Initialize state and coefficient vectors
63 z1.assign(numChannels, T(0));
64 a.assign(numChannels, T(0));
65 b.assign(numChannels, T(0));
66 togglePrepared = true;
67 }
68
76 void setByT60(size_t ch, Time<T> newT60Dc, Time<T> newT60Nyq, Time<T> newDelayTime) {
77 assert(ch < numChannels && "Channel index out of bounds.");
78 assert(sampleRate > T(0) && "Sample rate must be set and greater than zero.");
79
80 // Early exit if not prepared
81 if (!togglePrepared)
82 return;
83
84 // Clamp T60 values
85 T T60_DC = std::clamp(newT60Dc.toSeconds(sampleRate),
86 detail::DampingLimits<T>::MIN_T60,
87 detail::DampingLimits<T>::MAX_T60);
88 T T60_NYQ = std::clamp(newT60Nyq.toSeconds(sampleRate),
89 detail::DampingLimits<T>::MIN_T60,
90 detail::DampingLimits<T>::MAX_T60);
91
92 // Compute coefficients based on T60 values
93 T g0 = std::pow(T(10), -3.0 * newDelayTime.toSeconds(sampleRate) / T60_DC);
94 T g1 = std::pow(T(10), -3.0 * newDelayTime.toSeconds(sampleRate) / T60_NYQ);
95 a[ch] = (g0 + g1) / 2;
96 b[ch] = (g0 - g1) / 2;
97 }
98
106 T processSample(size_t ch, T x) {
107 assert(ch < numChannels && "Channel index out of bounds");
108 T y = a[ch] * x + b[ch] * z1[ch];
109 z1[ch] = y;
110 return y;
111 }
112
114 void reset() { std::fill(z1.begin(), z1.end(), T(0)); }
115
117 size_t getNumChannels() const { return numChannels; }
118
120 bool isPrepared() const { return togglePrepared; }
121
122 private:
123 bool togglePrepared = false;
124 T sampleRate = T(44100);
125 size_t numChannels = 0;
126 std::vector<T> a; // feedforward coefficient
127 std::vector<T> b; // feedback coefficient
128 std::vector<T> z1; // state per channel
129};

139template <typename T>
140class DampingFilter<T, DampingType::BiquadShelf> {
141 public:
143 DampingFilter() = default;
144
150 DampingFilter(size_t newNumChannels, T newSampleRate) { prepare(newNumChannels, newSampleRate); }
151
153 DampingFilter(const DampingFilter&) = delete;
154 DampingFilter& operator=(const DampingFilter&) = delete;
155 DampingFilter(DampingFilter&&) = delete;
156 DampingFilter& operator=(DampingFilter&&) = delete;
157
164 void prepare(size_t newNumChannels, T newSampleRate) {
165 numChannels = utils::detail::clampChannels(newNumChannels);
166 sampleRate = utils::detail::clampSampleRate(newSampleRate);
167 shelf.prepare(newNumChannels, newSampleRate, BiquadType::Highshelf);
168 gBase.resize(numChannels, T(0));
169 togglePrepared = true;
170 }
171
181 void
182 setByT60(size_t ch, Frequency<T> newCrossOverFreq, Time<T> newT60Low, Time<T> newT60High, Time<T> newDelayTime) {
183 assert(ch < numChannels && "Channel index out of bounds.");
184 assert(sampleRate > T(0) && "Sample rate must be set and greater than zero.");
185 // Early exit if not prepared
186 if (!togglePrepared)
187 return;
188
189 // Clamp crossover frequency
190 T crossOverFreqHz = std::clamp(newCrossOverFreq.toHertz(sampleRate),
191 detail::FilterLimits<T>::MIN_FREQ_NORM * sampleRate,
192 detail::FilterLimits<T>::MAX_FREQ_NORM * sampleRate);
193 // Clamp T60 values
194 T T60_Low = std::clamp(newT60Low.toSeconds(sampleRate),
195 detail::DampingLimits<T>::MIN_T60_SEC,
196 detail::DampingLimits<T>::MAX_T60_SEC);
197
198 T T60_High = std::clamp(newT60High.toSeconds(sampleRate),
199 detail::DampingLimits<T>::MIN_T60_SEC,
200 detail::DampingLimits<T>::MAX_T60_SEC);
201
202 // Set crossover frequency
203 shelf.setFreq(newCrossOverFreq);
204
205 // Compute gains based on T60
206 T gLow = std::pow(T(10), -3.0 * newDelayTime.toSeconds(sampleRate) / T60_Low);
207 T gHigh = std::pow(T(10), -3.0 * newDelayTime.toSeconds(sampleRate) / T60_High);
208 // Switch shelving type based on gain relationship
209 gHigh < gLow ? shelf.setType(BiquadType::Highshelf) // damp high frequencies if gHigh < gBase
210 : shelf.setType(BiquadType::Lowshelf); // damp low frequencies if gLow < gBase
211
212 // Base gain = max of the two gains i.e. least damping
213 gBase[ch] = std::max(gLow, gHigh);
214
215 // Compute shelf gain relative to base gain
216 T shelfGain = std::min(gLow, gHigh) / gBase[ch];
217 shelf.setGain(Gain<T>::Linear(shelfGain));
218 }
219
226 T processSample(size_t ch, T x) { return shelf.processSample(ch, gBase[ch] * x); }
227
229 void reset() { shelf.reset(); }
230
232 size_t getNumChannels() const { return numChannels; }
233
235 bool isPrepared() const { return togglePrepared; }
236
237 private:
238 bool togglePrepared = false;
239 T sampleRate = T(44100);
240 size_t numChannels = 0;
241 std::vector<T> gBase;
242 BiquadFilter<T> shelf;
243};
