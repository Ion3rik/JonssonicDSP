83 void processBlock(const T* const* dryInput,
84 const T* const* wetInput,
85 T* const* output,
86 size_t numSamples,
87 size_t dryDelaySamples = 0) {
88 for (size_t ch = 0; ch < numChannels; ++ch) {
89 for (size_t n = 0; n < numSamples; ++n) {
90 T mixValue = mix.getNextValue(ch);
91
92 // Equal-power crossfade: cos(x) for dry, sin(x) for wet
93 T dryGain = std::cos(mixValue * utils::pi_over_2<T>);
94 T wetGain = std::sin(mixValue * utils::pi_over_2<T>);
95
96 // Apply dry delay if needed
97 T drySample = dryInput[ch][n];
98 dryDelayBuffer.write(ch, drySample);
99 drySample = dryDelayBuffer.read(ch, dryDelaySamples);
100
101 // Mix dry and wet signals
102 output[ch][n] = drySample * dryGain + wetInput[ch][n] * wetGain;
103 }
104 }
