80 float processSample(size_t ch, T input) {
81 // SERIES ROUTING
82 if constexpr (RoutingType == Routing::Series) {
83 T output = input;
84 for (size_t s = 0; s < topology.getNumSections(); ++s) {
85 output = topology.processSample(ch, s, output);
86 }
87 return output;
88 }
89 // PARALLEL ROUTING
90 if constexpr (RoutingType == Routing::Parallel) {
91 T output = T(0);
92 for (size_t s = 0; s < topology.getNumSections(); ++s) {
93 output += topology.processSample(ch, s, input);
94 }
95 return output;
96 }
97 }

106 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
107 for (size_t ch = 0; ch < topology.getNumChannels(); ++ch)
108 for (size_t n = 0; n < numSamples; ++n)
109 output[ch][n] = processSample(ch, input[ch][n]);
110 }

299 // Function to apply design coefficients to the topology for a specific channel and section
300 void applyDesignToTopology(size_t ch, size_t section) {
301 assert(ch < numChannels && "Channel index out of bounds");
302 assert(section < numSections && "Section index out of bounds");
303 T b0, b1, a1;
304 design.computeCoeffs(ch, section, b0, b1, a1);
