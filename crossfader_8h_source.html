66 void processBlock(const T* const* input1,
67 const T* const* input2,
68 T* const* output,
69 size_t numSamples,
70 size_t input1DelaySamples = 0) {
71 for (size_t n = 0; n < numSamples; ++n) {
72 T crossFadeOn = static_cast<T>(crossfadeSamplePos < crossFadeTimeSamples);
73 T fadeFactor = static_cast<T>(crossfadeSamplePos) /
74 static_cast<T>(crossFadeTimeSamples - 1) * crossFadeOn +
75 (T(1) - crossFadeOn);
76
77 T g1 = std::cos(fadeFactor * pi_over_2<T>);
78 T g2 = std::sin(fadeFactor * pi_over_2<T>);
79
80 for (size_t ch = 0; ch < numChannels; ++ch) {
81 T input1sample = input1[ch][n];
82 latencyBuffer.write(ch, input1sample);
83 input1sample = latencyBuffer.read(ch, input1DelaySamples);
84
85 output[ch][n] = input1sample * g1 + input2[ch][n] * g2;
86 }
87 crossfadeSamplePos += static_cast<size_t>(crossFadeOn);
88 }
89 }
