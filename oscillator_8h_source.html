184 // Generate waveform sample at given phase (0.0 to 1.0)
185 inline T generateWaveform(T phase) const {
186 switch (waveform) {
187 case Waveform::Sine:
188 // Sine wave
189 return std::sin(T(2.0 * M_PI) * phase);
190
191 case Waveform::Saw:
192 // Sawtooth wave
193 return T(2.0) * phase - T(1.0);
194
195 case Waveform::Square:
196 // Square wave
197 return (phase < T(0.5)) ? T(-1.0) : T(1.0);
198
199 case Waveform::Triangle:
200 // Triangle wave
201 return T(1.0) - std::abs(T(4.0) * phase - T(2.0));
202
203 default:
204 return T(0);
205 }
206 }
