59 void prepare(size_t newNumChannels, size_t newMaxBlockSize, T newSampleRate) {
60 // Clamp and set number of channels and sample rate
61 numChannels = utils::detail::clampChannels(newNumChannels);
62 sampleRate = utils::detail::clampSampleRate(newSampleRate);
63
64 // Prepare oversampled processor if needed
65 if constexpr (OversamplingFactor > 1)
66 oversampledProcessor.prepare(numChannels, newMaxBlockSize);
67
68 // Prepare wave shaper processor at proper sample rate
69 waveShaper.prepare(numChannels, sampleRate * OversamplingFactor);
70
71 // Prepare pre- and post-filters if enabled
72 if constexpr (PreFilter)
73 preFilter.prepare(numChannels, sampleRate);
74
75 if constexpr (PostFilter)
76 postFilter.prepare(numChannels, sampleRate);
77 }

101 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
102
103 // PRE-FILTERING ENABLED SIGNAL PATH
104 if constexpr (PreFilter) {
105 preFilter.processBlock(input, output, numSamples);
106
107 // Oversampled waveshaping
108 if constexpr (OversamplingFactor > 1) {
109 oversampledProcessor.processBlock(output,
110 output,
111 numSamples,
112 [this](const T* const* in, T* const* out, size_t oversampledSamples) {
113 waveShaper.processBlock(in, out, oversampledSamples);
114 });
115
116 }
117 // No oversampling
118 else
119 waveShaper.processBlock(output, output, numSamples);
120 }
121
122 // PRE-FILTERING DISABLED SIGNAL PATH
123 if constexpr (!PreFilter) {
124 // Oversampled waveshaping
125 if constexpr (OversamplingFactor > 1) {
126 oversampledProcessor.processBlock(input,
127 output,
128 numSamples,
129 [this](const T* const* in, T* const* out, size_t oversampledSamples) {
130 waveShaper.processBlock(in, out, oversampledSamples);
131 });
132
133 }
134 // No oversampling
135 else
136 waveShaper.processBlock(input, output, numSamples);
137 }
138
139 // Post-filtering if enabled
140 if constexpr (PostFilter)
141 postFilter.processBlock(output, output, numSamples);
142 }
