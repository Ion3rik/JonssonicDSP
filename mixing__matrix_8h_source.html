100 void mix(const T* input, T* output) const {
101 // Hadamard transform using the Fast Walsh-Hadamard Transform (FWHT) algorithm
102 for (size_t i = 0; i < size; ++i)
103 output[i] = input[i];
104 for (size_t len = 1; len < size; len <<= 1) {
105 for (size_t i = 0; i < size; i += (len << 1)) {
106 for (size_t j = 0; j < len; ++j) {
107 T u = output[i + j];
108 T v = output[i + j + len];
109 output[i + j] = u + v;
110 output[i + j + len] = u - v;
111 }
112 }
113 }
114 // Apply normalization
115 for (size_t i = 0; i < size; ++i)
116 output[i] *= norm;
117 }
118
119 private:
120 size_t size;
121 T norm; // Normalization factor 1/sqrt(N)
122};

125// Householder specialization (reflection across a vector, here: all-ones vector)

127template <typename T>
128class MixingMatrix<T, MixingMatrixType::Householder> {
129 public:
131 MixingMatrix() = default;

140 ~MixingMatrix() = default;
141
146 void resize(size_t N) { size = N; }

155 void mix(const T* in, T* out) const {
156 T inputSum = T(0);
157 for (size_t i = 0; i < size; ++i) {
158 inputSum += in[i];
159 }
160 T coeff = T(2) / static_cast<T>(size) * inputSum;
161 for (size_t i = 0; i < size; ++i) {
162 out[i] = in[i] - coeff;
163 }
164 }

166 private:
167 size_t size;
168};

299 void fillDecorrelatedSum() {
300 size_t numInputs = denseMatrix.getNumInputs();
301 size_t numOutputs = denseMatrix.getNumOutputs();
302 T norm = static_cast<T>(1) / std::sqrt(static_cast<T>(numInputs));
303 for (size_t out = 0; out < numOutputs; ++out) {
304 for (size_t in = 0; in < numInputs; ++in) {
305 int sign = utils::parity_sign(static_cast<uint64_t>(in & out));
306 denseMatrix.set(out, in, norm * static_cast<T>(sign));
