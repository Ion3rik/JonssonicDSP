237 // Helper function to compute read index and fractional part for interpolation (in-place
238 // version)
239 void computeReadIndexAndFrac(T delay, size_t writeIdx, size_t& readIndex, T& delayFrac) const {
240 // Clamp delay to valid range as safety measure
241 delay = std::max(T(0), std::min(delay, static_cast<T>(bufferSize - 1)));
242
243 // Split into integer and fractional parts using floor for proper handling
244 T delayFloor = std::floor(delay);
245 size_t delayInt = static_cast<size_t>(delayFloor);
246 delayFrac = delay - delayFloor;
247
248 // Calculate read index with wrap-around
249 readIndex = (writeIdx + bufferSize - delayInt) & (bufferSize - 1);
250 }
