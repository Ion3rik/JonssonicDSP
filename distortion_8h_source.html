65 void prepare(size_t newNumChannels, size_t newMaxBlockSize, T newSampleRate) {
66
67 // Store global parameters
68 numChannels = utils::detail::clampChannels(newNumChannels);
69 sampleRate = utils::detail::clampSampleRate(newSampleRate);
70
71 // Prepare processing stages
72 distortion.prepare(newNumChannels, newMaxBlockSize, newSampleRate);
73 distortionOS.prepare(newNumChannels, newMaxBlockSize, newSampleRate);
74
75 // Setup post-filters for tone control
76 distortion.setPostFilterResponse(BiquadFilter<T>::Response::Lowpass);
77 distortionOS.setPostFilterResponse(BiquadFilter<T>::Response::Lowpass);
78
79 // Setup pre-filters to remove low-frequency content
80 distortion.setPreFilterResponse(BiquadFilter<T>::Response::Highpass);
81 distortionOS.setPreFilterResponse(BiquadFilter<T>::Response::Highpass);
82 distortion.setPreFilterFrequency(Frequency<T>::Hertz(T(PRE_FILTER_CUTOFF_HZ)));
83 distortionOS.setPreFilterFrequency(Frequency<T>::Hertz(T(PRE_FILTER_CUTOFF_HZ)));
84
85 // Prepare dry/wet mixer with latency compensation and output gain
86 size_t oversamplerLatencySamples = distortionOS.getLatencySamples();
87 dryWetMixer.prepare(newNumChannels, newSampleRate, oversamplerLatencySamples);
88
89 // Prepare internal buffers and oversampler
90 fxBuffer.resize(newNumChannels, newMaxBlockSize);
91
92 // Set parameter smoothing times
93 distortion.setControlSmoothingTime(Time<T>::Milliseconds(PARAM_SMOOTH_TIME_MS));
94 distortionOS.setControlSmoothingTime(Time<T>::Milliseconds(PARAM_SMOOTH_TIME_MS));
95 dryWetMixer.setControlSmoothingTime(Time<T>::Milliseconds(PARAM_SMOOTH_TIME_MS));
96
97 // Set Default Parameters
98 setDriveDb(T(0), true); // 0 dB drive
99 setAsymmetry(T(0), true); // 0 asymmetry
100 setShape(T(0.5), true); // Mid shape
101 setToneFrequency(T(12000)); // 12 kHz tone cutoff
102 setOutputGainDb(T(0), true); // 0 dB output gain
103 setOversamplingEnabled(false);
104 }
105
106 void reset() {
107 distortion.reset();
108 distortionOS.reset();
109 outputGain.reset();
110 dryWetMixer.reset();
111 fxBuffer.clear();
112 }

120 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
121 // Copy input to fxBuffer for processing
122 utils::copyToBuffer<T>(input, fxBuffer.writePtrs(), numChannels, numSamples);
123
124 // Process oversampled distortion if enabled
125 if (toggleOversampling) {
126 distortionOS.processBlock(fxBuffer.readPtrs(), fxBuffer.writePtrs(), numSamples);
127 }
128 // Process non-oversampled distortion otherwise
129 else {
130 distortion.processBlock(fxBuffer.readPtrs(), fxBuffer.writePtrs(), numSamples);
131 }
132
133 // Apply dry/wet mixing (with delay compensation if oversampling)
134 size_t dryDelaySamples = getLatencySamples();
135 dryWetMixer.processBlock(input, fxBuffer.readPtrs(), output, numSamples, dryDelaySamples);
136
137 // Apply output gain
138 outputGain.applyToBuffer(output, numSamples);
139 }
