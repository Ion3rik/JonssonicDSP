52 void prepare(size_t newNumChannels, T newSampleRate) {
53 // Clamp and set number of channels and sample rate

86 T processSample(size_t ch, T input) {
87 // Apply input gain
88 input *= inputGain.getNextValue(ch);
89 // Apply bias
90 input += bias.getNextValue(ch);
91 // Apply asymmetry (branchless)
92 T asym = asymmetry.getNextValue(ch);
93 T sign = (T(0) < input) - (input < T(0)); // 1 if input>0, -1 if input<0, 0 if input==0
94 input *= (T(1) + asym * sign);
95 // Apply waveshaping
96 input = shaper.processSample(input);
97 // Apply output gain
98 input *= outputGain.getNextValue(ch);
99 return input;
100 }

108 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
109 for (size_t ch = 0; ch < numChannels; ++ch) {
110 for (size_t n = 0; n < numSamples; ++n) {
111 // Get input sample
112 T sample = input[ch][n];
113 // Apply input gain
114 sample *= inputGain.getNextValue(ch);
115 // Apply bias
116 sample += bias.getNextValue(ch);
117 // Apply asymmetry (branchless)
118 T asym = asymmetry.getNextValue(ch);
119 T sign =
120 (T(0) < sample) - (sample < T(0)); // 1 if input>0, -1 if input<0, 0 if input==0
121 sample *= (T(1) + asym * sign);
122 // Apply waveshaping
123 output[ch][n] = shaper.processSample(sample, shape.getNextValue(ch));
124 // Apply output gain
125 output[ch][n] *= outputGain.getNextValue(ch);
126 }
127 }
