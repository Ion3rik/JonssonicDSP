23template <typename T>
24struct Time {
25 private:
26 Time(T v, TimeUnit u) : value(v), unit(u) {}
27
28 public:
29 T value;
30 TimeUnit unit;
31
33 static Time Samples(T v) { return Time(std::max(v, T(0)), TimeUnit::Samples); }
34 static Time Milliseconds(T v) { return Time(std::max(v, T(0)), TimeUnit::Milliseconds); }
35 static Time Seconds(T v) { return Time(std::max(v, T(0)), TimeUnit::Seconds); }
36
38 T toSamples(T sampleRate) const {
39 switch (unit) {
40 case TimeUnit::Samples:
41 return value;
42 case TimeUnit::Milliseconds:
43 return value * sampleRate * T(0.001);
44 case TimeUnit::Seconds:
45 return value * sampleRate;
46 default:
47 assert(false && "Unknown TimeUnit");
48 return 0;
49 }
50 }
52 T toSeconds(T sampleRate) const {
53 assert(sampleRate > T(0) && "Sample rate must be positive");
54 switch (unit) {
55 case TimeUnit::Samples:
56 return value * T(1) / sampleRate;
57 case TimeUnit::Milliseconds:
58 return value * T(0.001);
59 case TimeUnit::Seconds:
60 return value;
61 default:
62 assert(false && "Unknown TimeUnit");
63 return T(0);
64 }
65 }
66
68 T toMilliseconds(T sampleRate) const {
69 assert(sampleRate > T(0) && "Sample rate must be positive");
70 switch (unit) {
71 case TimeUnit::Samples:
72 return value * T(1000) * T(1) / sampleRate;
73 case TimeUnit::Milliseconds:
74 return value;
75 case TimeUnit::Seconds:
76 return value * T(1000);
77 default:
78 assert(false && "Unknown TimeUnit");
79 return T(0);

169enum class GainUnit { Linear, Decibels };
170
172template <typename T>
173struct Gain {
174 private:
175 Gain(T v, GainUnit u) : value(v), unit(u) {}
176
177 public:
178 T value;
179 GainUnit unit;
180
182 static Gain Linear(T v) {
183 T sign = (v < T(0)) ? T(-1) : T(1);
184 T mag = std::max(std::abs(v), std::numeric_limits<T>::epsilon());
185 return Gain(sign * mag, GainUnit::Linear);
186 }
187 static Gain Decibels(T v) { return Gain(v, GainUnit::Decibels); }
188
190 T toLinear() const {
191 switch (unit) {
192 case GainUnit::Linear:
193 return value;
194 case GainUnit::Decibels:
195 return utils::dB2Mag(value);
196 default:
197 assert(false && "Unknown GainUnit");
198 return T(1);
199 }
200 }
208 T toDecibels() const {
209 switch (unit) {
210 case GainUnit::Linear:
211 return utils::mag2dB(std::abs(value));
212 case GainUnit::Decibels:
213 return value;
214 default:
215 assert(false && "Unknown GainUnit");
216 return T(0);
217 }
218 }
219
221 Gain operator-() const { return Gain(-value, unit); }
222};
