99 T processSample(size_t ch, T input) {
100 // Get smoothed gain values
101 T ffGain = feedforwardGain.getNextValue(ch);
102 T fbGain = feedbackGain.getNextValue(ch);
103
104 // Read delayed signal
105 T delayed = delayLine.readSample(ch);
106
107 // Compute what to write back (input + delayed feedback)
108 T toWrite = input + delayed * fbGain;
109 delayLine.writeSample(ch, toWrite);
110
111 // Compute output (input + delayed feedforward)
112 T output = input + delayed * ffGain;
113
114 return output;
115 }

149 void processBlock(const T* const* input, T* const* output, size_t numSamples) {

152 // Get smoothed gain values
