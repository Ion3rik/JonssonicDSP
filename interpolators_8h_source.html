84template <typename T>
85struct LagrangeInterpolator {
86 // For delay lines - interpolate backward in time
87 // Uses points: x[-3], x[-2], x[-1], x[0] where x[0] is the current index
88 // Only uses past samples to avoid causality issues
89 static T interpolateBackward(const T* buffer, size_t idx, float frac, size_t bufferSize) {
90 // Get 4 points: all in the past
91 size_t idx0 = idx; // x[0]
92 size_t idxM1 = (idx + bufferSize - 1) & (bufferSize - 1); // x[-1]
93 size_t idxM2 = (idx + bufferSize - 2) & (bufferSize - 1); // x[-2]
94 size_t idxM3 = (idx + bufferSize - 3) & (bufferSize - 1); // x[-3]
95
96 T x0 = buffer[idx0];
97 T xM1 = buffer[idxM1];
98 T xM2 = buffer[idxM2];
99 T xM3 = buffer[idxM3];
100
101 // Lagrange basis polynomials for fractional position
102 float fracSq = frac * frac;
103 float fracCube = fracSq * frac;
104
105 // L_0(frac) = -(frac - 1) * (frac - 2) * (frac - 3) / 6
106 T c0 = x0 * (-fracCube + 6.0f * fracSq - 11.0f * frac + 6.0f) / 6.0f;
107
108 // L_{-1}(frac) = frac * (frac - 2) * (frac - 3) / 2
109 T c1 = xM1 * (fracCube - 5.0f * fracSq + 6.0f * frac) / 2.0f;
110
111 // L_{-2}(frac) = -frac * (frac - 1) * (frac - 3) / 2
112 T c2 = xM2 * (-fracCube + 4.0f * fracSq - 3.0f * frac) / 2.0f;
113
114 // L_{-3}(frac) = frac * (frac - 1) * (frac - 2) / 6
115 T c3 = xM3 * (fracCube - 3.0f * fracSq + 2.0f * frac) / 6.0f;
116
117 return c0 + c1 + c2 + c3;
118 }
119
120 // For wavetables/resampling - interpolate forward
121 // Uses points: x[-1], x[0], x[1], x[2] where x[0] is the current index
122 static T interpolateForward(const T* buffer, size_t idx, float frac, size_t bufferSize) {
123 // Get 4 points: one before, current, and two ahead
124 size_t idx0 = idx;
125 size_t idxM1 = (idx + bufferSize - 1) & (bufferSize - 1); // x[-1]
126 size_t idxP1 = (idx + 1) & (bufferSize - 1); // x[+1]
127 size_t idxP2 = (idx + 2) & (bufferSize - 1); // x[+2]
128
129 T xM1 = buffer[idxM1];
130 T x0 = buffer[idx0];
131 T xP1 = buffer[idxP1];
132 T xP2 = buffer[idxP2];
133
134 // Lagrange basis polynomials for fractional position
135 float fracSq = frac * frac;
136 float fracCube = fracSq * frac;
137
138 // L_{-1}(frac) = -frac * (frac - 1) * (frac - 2) / 6
139 T c0 = xM1 * (-fracCube + 3.0f * fracSq - 2.0f * frac) / 6.0f;
140
141 // L_0(frac) = (frac + 1) * (frac - 1) * (frac - 2) / 2
142 T c1 = x0 * (fracCube - 2.0f * fracSq - frac + 2.0f) / 2.0f;
143
144 // L_1(frac) = -(frac + 1) * frac * (frac - 2) / 2
145 T c2 = xP1 * (-fracCube + fracSq + 2.0f * frac) / 2.0f;
146
147 // L_2(frac) = (frac + 1) * frac * (frac - 1) / 6
148 T c3 = xP2 * (fracCube - fracSq) / 6.0f;
149
150 return c0 + c1 + c2 + c3;
151 }
152};
