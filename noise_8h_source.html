52 void prepare(size_t newNumChannels) {
53 numChannels = newNumChannels;
54 rngs.resize(numChannels);
55 hasSpare.resize(numChannels, false);
56 spare.resize(numChannels, 0.0f);
57 for (size_t ch = 0; ch < numChannels; ++ch) {
58 rngs[ch].seed(static_cast<uint32_t>(2463534242UL + ch * 7919));
59 hasSpare[ch] = false;
60 }
61 }

66 void reset() {
67 for (size_t ch = 0; ch < numChannels; ++ch) {
68 rngs[ch].seed(static_cast<uint32_t>(2463534242UL + ch * 7919));
69 hasSpare[ch] = false;
70 }
71 }

76 T processSample(size_t ch) {
77 // Gaussian white noise sample using Box-Muller transform
78 if (hasSpare[ch]) {
79 hasSpare[ch] = false;
80 return T(spare[ch]);
81 }
82 float u, v, s;
83 do {
84 u = rngs[ch].nextFloat();
85 v = rngs[ch].nextFloat();
86 s = u * u + v * v;
87 } while (s >= 1.0f || s == 0.0f);
88 float mul = std::sqrt(-2.0f * std::log(s) / s);
89 spare[ch] = v * mul;
90 hasSpare[ch] = true;
91 return T(u * mul);
92 }
93
97 void processBlock(T* const* output, size_t numSamples) {
98 for (size_t ch = 0; ch < numChannels; ++ch) {
99 for (size_t n = 0; n < numSamples; ++n) {
100 output[ch][n] = processSample(ch);
101 }
102 }
103 }
104
105 private:
106 size_t numChannels = 0;
107 std::vector<utils::Xorshift32> rngs;
108 std::vector<bool> hasSpare;
109 std::vector<float> spare;
110};
