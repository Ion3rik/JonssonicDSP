63 void prepare(size_t newNumChannels, T newSampleRate) {
64 // Store global parameters
65 numChannels = utils::detail::clampChannels(newNumChannels);
66 sampleRate = utils::detail::clampSampleRate(newSampleRate);
67
68 // Prepare DSP components
69 multiTapDelay.prepare(numChannels, sampleRate, Time<T>::Milliseconds(MAX_DELAY_MS));
70 multiTapDelay.setControlSmoothingTime(Time<T>::Milliseconds(T(SMOOTHING_TIME_MS)));
71 lfo.prepare(numChannels * NUM_VOICES, sampleRate);
72 lfo.setWaveform(Waveform::Sine);
73
74 // Prepare parameters
75 modDepthSamples.prepare(numChannels, sampleRate);
76 lfoPhaseOffset.prepare(numChannels * NUM_VOICES, sampleRate);
77 feedback.prepare(numChannels, sampleRate);
78
79 // Set voice gains
80 Gain<T> voiceGain = Gain<T>::Linear(NORM_FACTOR);
81 for (size_t tap = 0; tap < NUM_VOICES; ++tap)

84 // Set Default Parameters

86 setDepth(T(0.5), true);
87 setFeedback(T(0.0), true);
88 setDelayMs(T(15.0), true);

106 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
107 for (size_t ch = 0; ch < numChannels; ++ch) {
108 // Calculate the base LFO index for this channel
109 size_t voiceBaseIdx = ch * NUM_VOICES;
110 for (size_t n = 0; n < numSamples; ++n) {
111 // Get current input sample
112 T inputSample = input[ch][n];
113
114 // Initialize output to zero before accumulating
115 T outputSample = T(0);
116
117 // Get the modulation depth for this channel
118 T modDepth = modDepthSamples.getNextValue(ch);
119
120 // Loop over active voices
121 for (size_t tap = 0; tap < NUM_VOICES; ++tap) {
122 // Get the phase offset for this channel-tap combination
123 T phaseOffset = lfoPhaseOffset.getNextValue(index(ch, tap));
124
125 // Run LFO for this channel-tap combination to get modulation value (unipolar: 0 to +1)
126 T lfoValue = lfo.processSample(voiceBaseIdx + tap, phaseOffset) * T(0.5) + T(0.5);
127 T mod = modDepth * lfoValue;
128
129 // Read delayed sample with interpolation and gain for this tap and accumulate to output
130 outputSample += multiTapDelay.readSample(ch, tap, mod);
131 }
132
133 // Write the input sample plus feedback into the delay line
134 multiTapDelay.writeSample(ch, inputSample + feedback.getNextValue(ch) * outputSample * NORM_FACTOR);
135 // Store the final output sample
136 output[ch][n] = outputSample;
137 }
138 }
139 }

179 void setSpread(T spread, bool skipSmoothing = false) {
180 // Distribute voices
181 T voiceSpread = T(0.5) / static_cast<T>(NUM_VOICES);
182
183 // Spread param offsets channels further apart
184 for (size_t ch = 0; ch < numChannels; ++ch) {
185 for (size_t tap = 0; tap < NUM_VOICES; ++tap) {
186 T offset = (spread * ch) / static_cast<T>(numChannels) + (tap * voiceSpread);
187 lfoPhaseOffset.setTarget(index(ch, tap), offset, skipSmoothing);
188 }
189 }
190 }
