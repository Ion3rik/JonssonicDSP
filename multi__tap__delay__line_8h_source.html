51 void prepare(size_t newNumChannels, T newSampleRate, Time<T> newMaxDelay) {
52 // Store config parameters with clamping
53 numChannels = utils::detail::clampChannels(newNumChannels);
54 sampleRate = utils::detail::clampSampleRate(newSampleRate);
55
56 // Prepare circular buffer based on max delay time
57 size_t maxDelaySamples = newMaxDelay.toSamples(sampleRate); // convert to samples
58 circularBuffer.resize(numChannels, maxDelaySamples); // resize circular buffer
59
60 // Prepare DSP parameters for tap delay times and gains
61 tapDelay.prepare(numChannels * NumTaps, sampleRate);
62 tapGain.prepare(numChannels * NumTaps, sampleRate);
63
64 // Set default bounds for delay times and gains
65 tapDelay.setBounds(T(0), maxDelaySamples);
66 tapGain.setBounds(T(-1), T(1));
67
68 // Mark as prepared
69 togglePrepared = true;
70 }

86 T processSample(size_t ch, T input) {
87 // Write input sample to buffer
88 circularBuffer.write(ch, input);
89
90 // Accumulate output from all taps
91 T output = T(0);
92 for (size_t tap = 0; tap < NumTaps; ++tap)
93 output += tapGain.getNextValue(index(ch, tap)) *
94 Interpolator::interpolate(circularBuffer, ch, tapDelay.getNextValue(index(ch, tap)));
95
96 // Return the mixed output from all taps
97 return output;
98 }

108 T processSample(size_t ch, T input, const std::array<T, NumTaps>& modulation) {
109 // Write input sample to buffer
110 circularBuffer.write(ch, input);
111
112 // Accumulate output from all taps with modulated delay
113 T output = T(0);
114 for (size_t tap = 0; tap < NumTaps; ++tap) {
115 // Calculate modulated delay time with internal clamping (base delay + modulation)
116 T modulatedDelay = tapDelay.applyAdditiveMod(index(ch, tap), modulation[tap]);
117
118 // Accumulate output from this tap with gain
119 output +=
120 tapGain.getNextValue(index(ch, tap)) * Interpolator::interpolate(circularBuffer, ch, modulatedDelay);
121 }
122
123 return output;
124 }
