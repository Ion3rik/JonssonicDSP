96 void prepare(size_t newNumChannels, T newSampleRate) {
97 // Store global parameters
98 numChannels = utils::detail::clampChannels(newNumChannels);
99 sampleRate = utils::detail::clampSampleRate(newSampleRate);
100
101 // Compute maximum delay time for FDN delay lines
102 T maxDelaySamples = T(0);
103 for (size_t d = 0; d < FDN_SIZE; ++d)
104 maxDelaySamples =
105 std::max(maxDelaySamples, MAX_DELAY_SCALE * static_cast<T>(FDNBaseDelays<FDN_SIZE>::values[d]));
106
107 // Prepare DSP components
108 preDelay.prepare(numChannels, sampleRate, Time<T>::Milliseconds(MAX_PRE_DELAY_MS));
109 fdn.prepare(numChannels, sampleRate, Time<T>::Samples(maxDelaySamples));
110 lowCutFilter.prepare(numChannels, sampleRate);
111 lowCutFilter.setResponse(BiquadFilter<T>::Response::Highpass);
112
113 // Set fixed parameters
114 fdn.setControlSmoothingTime(Time<T>::Milliseconds(SMOOTHING_TIME_MS));
115
116 // Initialize parameters to defaults
117 setReverbTimeLowS(T(2.0), true);
118 setReverbTimeHighS(T(1.0), true);
119 setDiffusion(T(0.5), true);
120 setPreDelayTimeMs(T(0.0), true);
121 setLowCutFreqHz(T(1000.0));
122 setModulationRateHz(T(1.0));
123 setModulationDepth(T(0.1));
124 }

186 void setDiffusion(T diffusionAmount, bool skipSmoothing = false) {
187 // Map the diffusion parameter to delay length scale (MIN_DELAY_SCALE .. MAX_DELAY_SCALE)
188 T scaledDiffusion = std::clamp((1 - diffusionAmount) * (MAX_DELAY_SCALE - MIN_DELAY_SCALE) + MIN_DELAY_SCALE,
189 MIN_DELAY_SCALE,
190 MAX_DELAY_SCALE);
191
192 for (size_t m = 0; m < FDN_SIZE; ++m) {
193 // Compute scaled delay lengths
194 size_t delaySamples = static_cast<size_t>(FDNBaseDelays<FDN_SIZE>::values[m] * scaledDiffusion);
195 fdn.setDelay(m, Time<T>::Samples(delaySamples), skipSmoothing);
196 }
197 }
