138 void updateCoeffs() {
139 // Early exit if not prepared
140 if (!togglePrepared)
141 return;

143 // Initialize coefficients variables
144 T b0 = T(0), b1 = T(0), a1 = T(0);
145
146 // Compute coefficients based on filter type
147 switch (type) {
148 case FirstOrderType::Lowpass:
149 jnsc::detail::computeFirstOrderLowpassCoeffs<T>(freqNormalized, b0, b1, a1);
150 FirstOrderCore.setSectionCoeffs(0, b0, b1, a1);
151 break;
152 case FirstOrderType::Highpass:
153 jnsc::detail::computeFirstOrderHighpassCoeffs<T>(freqNormalized, b0, b1, a1);
154 FirstOrderCore.setSectionCoeffs(0, b0, b1, a1);
155 break;
156 case FirstOrderType::Allpass:
157 jnsc::detail::computeFirstOrderAllpassCoeffs<T>(freqNormalized, b0, b1, a1);
158 FirstOrderCore.setSectionCoeffs(0, b0, b1, a1);
159 break;
160 case FirstOrderType::Lowshelf:
161 jnsc::detail::computeFirstOrderLowshelfCoeffs<T>(freqNormalized, gain, b0, b1, a1);
162 FirstOrderCore.setSectionCoeffs(0, b0, b1, a1);
163 break;
164 case FirstOrderType::Highshelf:
165 jnsc::detail::computeFirstOrderHighshelfCoeffs<T>(freqNormalized, gain, b0, b1, a1);
166 FirstOrderCore.setSectionCoeffs(0, b0, b1, a1);
167 break;
