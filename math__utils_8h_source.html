58 n |= n >> 1;

71// Compute the DFT of a real input vector (output is complex)
72template <typename T>
73std::vector<std::complex<T>> complexSpectrum(const std::vector<T>& input) {
74 size_t N = input.size();
75 std::vector<std::complex<T>> output(N);
76 for (size_t k = 0; k < N; ++k) {
77 std::complex<T> sum(0, 0);
78 for (size_t n = 0; n < N; ++n) {
79 T angle = -2 * pi<T> * k * n / N;
80 sum += input[n] * std::exp(std::complex<T>(0, angle));
81 }
82 output[k] = sum;
83 }
84 return output;
85}

87// Compute the magnitude spectrum of a real input vector
88// If oneSided=true, returns only the first N/2+1 bins (for real input)
89// If dB=true, returns 20*log10(mag+1e-12)
90template <typename T>
91std::vector<T>
92magnitudeSpectrum(const std::vector<T>& input, bool oneSided = false, bool dB = false) {
93 auto spec = complexSpectrum(input);
94 size_t N = spec.size();
95 size_t outLen = oneSided ? (N / 2 + 1) : N;
96 std::vector<T> mag(outLen);
97 constexpr T minMag = T(1e-12); // minimum magnitude to avoid log(0)
98 constexpr T minDb = T(-120); // minimum dB value to clamp to
99 for (size_t i = 0; i < outLen; ++i) {
100 T val = std::abs(spec[i]);
101 if (dB) {
102 val = 20 * std::log10(std::max(val, minMag));
103 if (val < minDb)
104 val = minDb;
105 }
106 mag[i] = val;
107 }
108 return mag;
109}

161template <typename T>
162std::pair<std::vector<T>, std::vector<int>>
163xcorr(const std::vector<T>& x, const std::vector<T>& y, bool normalize = false) {
164 size_t N = x.size();
165 size_t M = y.size();
166 size_t len = N + M - 1;
167 std::vector<T> result(len, T(0));
168 std::vector<int> lags(len);
169
170 // Generate lag indices: -(M-1) to (N-1)
171 int startLag = -(static_cast<int>(M) - 1);
172 for (size_t i = 0; i < len; ++i) {
173 lags[i] = startLag + static_cast<int>(i);
174 }
175
176 for (size_t n = 0; n < len; ++n) {
177 for (size_t k = 0; k < M; ++k) {
178 if (n >= k && (n - k) < N) {
179 result[n] += x[n - k] * y[k];
180 }
181 }
182 }
183
184 if (normalize) {
185 T maxVal = T(0);
186 for (const auto& val : result) {
187 if (std::abs(val) > maxVal) {
188 maxVal = std::abs(val);
189 }
190 }
191 if (maxVal > T(0)) {
192 for (auto& val : result) {
193 val /= maxVal;
194 }
195 }
196 }
197
198 return {result, lags};
199}

210template <typename T>
211int measureLatency(const std::vector<T>& reference, const std::vector<T>& delayed) {
212 auto [corr, lags] = xcorr(reference, delayed, true);
213 T maxVal = T(0);
214 size_t maxIdx = 0;
215 for (size_t i = 0; i < corr.size(); ++i) {
216 if (corr[i] > maxVal) {
217 maxVal = corr[i];
218 maxIdx = i;
219 }
220 }
221 int rawLag = lags[maxIdx];
222 // Convert from xcorr convention to delay: negate and adjust for signal length
223 int delay = (static_cast<int>(reference.size()) - 1) + rawLag;
224 return delay;
225}

227// Returns 1 if the number of set bits in x is even, -1 if odd (cross-platform)
228inline int parity_sign(uint64_t x) {
229 x ^= x >> 32;
230 x ^= x >> 16;
231 x ^= x >> 8;
232 x ^= x >> 4;
233 x ^= x >> 2;
234 x ^= x >> 1;

241struct Xorshift32 {
242 uint32_t state;
243 Xorshift32(uint32_t seed = 2463534242UL) : state(seed) {}
244 void seed(uint32_t s) { state = s ? s : 2463534242UL; }
245 uint32_t next() {
246 uint32_t x = state;
247 x ^= x << 13;
248 x ^= x >> 17;
249 x ^= x << 5;
250 state = x;
251 return x;
252 }
