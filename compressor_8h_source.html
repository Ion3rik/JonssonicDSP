57 void prepare(size_t newNumChannels, T newSampleRate) {

59 // Store global parameters

63 // Prepare compressor components

71 compressor.setControlSmoothingTime(Time<T>::Milliseconds(CONTROL_SMOOTH_TIME_MS));
