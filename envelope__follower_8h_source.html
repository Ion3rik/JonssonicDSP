88 T processSample(size_t ch, T input) {
89 // Rectify input (i.e. find peak level)
90 T rectified = std::abs(input);
91
92 // What stage are we in?
93 T inAttack = static_cast<T>(rectified > envelope[ch]);
94 T inRelease = T(1) - inAttack;
95
96 // Compute the smoothing coefficient
97 T coeff =
98 inAttack * attackCoeff.getNextValue(ch) + inRelease * releaseCoeff.getNextValue(ch);
99
100 // Apply smoothing to the rectified value
101 envelope[ch] += coeff * (rectified - envelope[ch]);
102
103 // Return the envelope value
104 return envelope[ch];
105 }

113 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
114 for (size_t ch = 0; ch < numChannels; ++ch)
115 for (size_t n = 0; n < numSamples; ++n) {
116 {
117 output[ch][n] = processSample(ch, input[ch][n]);
118 }
119 }
120 }

201 void updateCoefficients(bool skipSmoothing) {
202
203 // Set target coefficients based on current attack and release times
204 attackCoeff.setTarget(1.0 - std::exp(-1.0 / (params.attackTimeSec * sampleRate)),
205 skipSmoothing);
206 releaseCoeff.setTarget(1.0 - std::exp(-1.0 / (params.releaseTimeSec * sampleRate)),
207 skipSmoothing);
