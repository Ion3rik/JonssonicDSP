56 void prepare(size_t newNumChannels, size_t newNumSections, T newSampleRate) {
57 // Initialize parameters
58 numChannels = utils::detail::clampChannels(newNumChannels);
59 sampleRate = utils::detail::clampSampleRate(newSampleRate);
60 numSections = std::clamp(newNumSections, size_t(1), detail::FilterLimits<T>::MAX_SECTIONS);
61
62 // Initialize per-section parameter vectors
63 freqNormalized.assign(numSections, T(0.25)); // default to quarter Nyquist
64 Q.assign(numSections, T(0.707)); // default to Butterworth
65 gain.assign(numSections, T(1)); // default to unity gain
66 type.assign(numSections, BiquadType::Lowpass); // default to lowpass
67
68 // Prepare the underlying biquad core processor
69 biquadCore.prepare(numChannels, numSections);
70 togglePrepared = true;
71
72 // Update coefficients for all sections
73 updateCoeffs();
74 }

174 void updateCoeffs(size_t section) {
175 // Early exit if not prepared
176 if (!(togglePrepared))
177 return;
178 // Initialize coefficients variables
179 T b0 = T(0), b1 = T(0), b2 = T(0), a1 = T(0), a2 = T(0);
180 switch (type[section]) {
181 case BiquadType::Lowpass:
182 detail::computeLowpassCoeffs<T>(freqNormalized[section], Q[section], b0, b1, b2, a1, a2);
183 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
184 break;
185 case BiquadType::Highpass:
186 detail::computeHighpassCoeffs<T>(freqNormalized[section], Q[section], b0, b1, b2, a1, a2);
187 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
188 break;
189 case BiquadType::Bandpass:
190 detail::computeBandpassCoeffs<T>(freqNormalized[section], Q[section], b0, b1, b2, a1, a2);
191 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
192 break;
193 case BiquadType::Allpass:
194 detail::computeAllpassCoeffs<T>(freqNormalized[section], Q[section], b0, b1, b2, a1, a2);
195 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
196 break;
197 case BiquadType::Notch:
198 detail::computeNotchCoeffs<T>(freqNormalized[section], Q[section], b0, b1, b2, a1, a2);
199 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
200 break;
201 case BiquadType::Peak:
202 detail::computePeakCoeffs<T>(freqNormalized[section], Q[section], gain[section], b0, b1, b2, a1, a2);
203 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
204 break;
205 case BiquadType::Lowshelf:
206 detail::computeLowshelfCoeffs<T>(freqNormalized[section], Q[section], gain[section], b0, b1, b2, a1, a2);
207 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
208 break;
209 case BiquadType::Highshelf:
210 detail::computeHighshelfCoeffs<T>(freqNormalized[section], Q[section], gain[section], b0, b1, b2, a1, a2);
211 biquadCore.setSectionCoeffs(section, b0, b1, b2, a1, a2);
212 break;
213 default:
214 // Handle unsupported types or add implementations
215 break;
216 }
217 }
