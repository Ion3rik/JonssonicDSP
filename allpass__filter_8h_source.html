97 T processSample(size_t ch, T input) {
98 // Get smoothed gain value
99 T gainValue = gain.getNextValue(ch);
100
101 // Read delayed feedback state
102 T delayed = delayLine.readSample(ch);
103
104 // Compute output
105 T output = gainValue * input + delayed;
106
107 // Compute and write new feedback state
108 T newFeedback = input - gainValue * output;
109 delayLine.writeSample(ch, newFeedback);
110
111 return output;
112 }
