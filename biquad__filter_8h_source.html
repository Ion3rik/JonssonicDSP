73 float processSample(size_t ch, T input) {
74 // SERIES ROUTING
75 if constexpr (RoutingType == Routing::Series) {
76 T output = input;
77 for (size_t s = 0; s < topology.getNumSections(); ++s) {
78 output = topology.processSample(ch, s, output);
79 }
80 return output;
81 }
82 // PARALLEL ROUTING
83 if constexpr (RoutingType == Routing::Parallel) {
84 T output = T(0);
85 for (size_t s = 0; s < topology.getNumSections(); ++s) {
86 output += topology.processSample(ch, s, input);
87 }
88 return output;
89 }
90 }
91
99 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
100 for (size_t ch = 0; ch < topology.getNumChannels(); ++ch)
101 for (size_t n = 0; n < numSamples; ++n)
102 output[ch][n] = processSample(ch, input[ch][n]);
103 }
