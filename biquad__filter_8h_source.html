155 void updateCoeffs() {
156 // Early exit if not prepared
157 if (!(togglePrepared))
158 return;
159
160 // Initialize coefficients variables
161 T b0 = T(0), b1 = T(0), b2 = T(0), a1 = T(0), a2 = T(0);
162
163 // Compute coefficients based on filter type
164 switch (type) {
165 case BiquadType::Lowpass:
166 detail::computeLowpassCoeffs<T>(freqNormalized, Q, b0, b1, b2, a1, a2);
167 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
168 break;
169 case BiquadType::Highpass:
170 detail::computeHighpassCoeffs<T>(freqNormalized, Q, b0, b1, b2, a1, a2);
171 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
172 break;
173 case BiquadType::Bandpass:
174 detail::computeBandpassCoeffs<T>(freqNormalized, Q, b0, b1, b2, a1, a2);
175 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
176 break;
177 case BiquadType::Allpass:
178 detail::computeAllpassCoeffs<T>(freqNormalized, Q, b0, b1, b2, a1, a2);
179 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
180 break;
181 case BiquadType::Notch:
182 detail::computeNotchCoeffs<T>(freqNormalized, Q, b0, b1, b2, a1, a2);
183 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
184 break;
185 case BiquadType::Peak:
186 detail::computePeakCoeffs<T>(freqNormalized, Q, gain, b0, b1, b2, a1, a2);
187 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
188 break;
189 case BiquadType::Lowshelf:
190 detail::computeLowshelfCoeffs<T>(freqNormalized, Q, gain, b0, b1, b2, a1, a2);
191 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
192 break;
193 case BiquadType::Highshelf:
194 detail::computeHighshelfCoeffs<T>(freqNormalized, Q, gain, b0, b1, b2, a1, a2);
195 biquadCore.setSectionCoeffs(0, b0, b1, b2, a1, a2);
196 break;
