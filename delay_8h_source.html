61 void prepare(size_t newNumChannels, size_t newMaxBlockSize, T newSampleRate) {
62 // Store global parameters
63 numChannels = utils::detail::clampChannels(newNumChannels);
64 sampleRate = utils::detail::clampSampleRate(newSampleRate);
65
66 // Prepare Modulated Delay Stage
67 modulatedDelayStage.prepare(numChannels, Time<T>::Milliseconds(MAX_DELAY_MS), sampleRate);
68 modulatedDelayStage.setControlSmoothingTime(Time<T>::Milliseconds(T(SMOOTHING_TIME_MS)));
69
70 // Prepare Wow LFO
71 wowLfo.prepare(newNumChannels, newSampleRate);
72 wowLfo.setControlSmoothingTime(Time<T>::Milliseconds(T(SMOOTHING_TIME_MS)));
73 wowLfo.setWaveform(Waveform::Sine);
74 wowLfo.setFrequency(Frequency<T>::Hertz(0.3)); // 0.3 Hz for wow
75
76 // Prepare Flutter LFO
77 flutterLfo.prepare(newNumChannels, newSampleRate);

80 flutterLfo.setFrequency(Frequency<T>::Hertz(6.0)); // 6 Hz for flutter

106 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
107 // Process LFOs to modulation buffer (note: could implement oscillator bank to internalize additive oscillators)
108 for (size_t ch = 0; ch < numChannels; ++ch) {
109 for (size_t n = 0; n < numSamples; ++n) {
110 T wowValue = wowLfo.processSample(ch);
111 T flutterValue = flutterLfo.processSample(ch);
112 modulationBuffer[ch][n] =
113 wowValue * WOW_PORTION_OF_MODULATION + flutterValue * (T(1) - WOW_PORTION_OF_MODULATION);
114 }
115 }
116
117 // Process delay stage with external modulation
118 modulatedDelayStage.processBlock(input, output, modulationBuffer.readPtrs(), numSamples);
119 }
