100 template <typename ProcessFunc>
101 void processBlock(int factor,
102 const T* const* input,
103 T* const* output,
104 size_t numSamples,
105 ProcessFunc&& processFunc) {
106 switch (factor) {
107 case 1: // No oversampling
108 processFunc(input, output, numSamples);
109 break;
110
111 case 2: // 2x oversampling
112 processWithOversampling(oversampler2x,
113 input,
114 output,
115 numSamples,
116 std::forward<ProcessFunc>(processFunc));
117 break;
118
119 case 4: // 4x oversampling
120 processWithOversampling(oversampler4x,
121 input,
122 output,
123 numSamples,
124 std::forward<ProcessFunc>(processFunc));
125 break;
126
127 case 8: // 8x oversampling
128 processWithOversampling(oversampler8x,
129 input,
130 output,
131 numSamples,
132 std::forward<ProcessFunc>(processFunc));
133 break;
134
135 case 16: // 16x oversampling
136 processWithOversampling(oversampler16x,
137 input,
138 output,
139 numSamples,
140 std::forward<ProcessFunc>(processFunc));
141 break;
142
143 default: // Fallback to no oversampling
144 processFunc(input, output, numSamples);
145 break;
146 }
147 }
