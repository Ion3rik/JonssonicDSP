24template <typename T>
25inline void mapChannels(const T* const* input,
26 T* const* output,
27 size_t numInputChannels,
28 size_t numOutputChannels,
29 size_t numSamples) {
30 for (int outCh = 0; outCh < numOutputChannels; ++outCh) {
31 if (numInputChannels == numOutputChannels) {
32 // Direct copy
33 std::memcpy(output[outCh], input[outCh], sizeof(T) * numSamples);
34 } else if (numInputChannels > numOutputChannels) {
35 // Downmix: average groups of input channels
36 int groupSize = numInputChannels / numOutputChannels;
37 for (int n = 0; n < numSamples; ++n) {
38 T sum = 0;
39 for (int i = 0; i < groupSize; ++i)
40 sum += input[outCh * groupSize + i][n];
41 output[outCh][n] = sum / groupSize;
42 }
43 } else {
44 // Upmix: wrap or duplicate
45 int inCh = outCh % numInputChannels;
46 std::memcpy(output[outCh], input[inCh], sizeof(T) * numSamples);
47 }
48 }
49}

118template <typename T>
119void crossfadeBuffers(const T* const* bufferA,
120 const T* const* bufferB,
121 T* const* output,
122 size_t numChannels,
123 size_t numSamples) {
124 for (size_t ch = 0; ch < numChannels; ++ch) {
125 for (size_t n = 0; n < numSamples; ++n) {
126 // Equal power crossfade
127 T fadeFactor = static_cast<T>(n) / static_cast<T>(numSamples - 1);
128 T gainA =
129 std::cos(fadeFactor * (static_cast<T>(M_PI) / static_cast<T>(2))); // cos(0 to π/2)
130 T gainB =
131 std::sin(fadeFactor * (static_cast<T>(M_PI) / static_cast<T>(2))); // sin(0 to π/2)
132 output[ch][n] = bufferA[ch][n] * gainA + bufferB[ch][n] * gainB;
133 }
134 }
