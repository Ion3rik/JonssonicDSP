24template <typename T>
25inline void mapChannels(
26 const T* const* input, T* const* output, size_t numInputChannels, size_t numOutputChannels, size_t numSamples) {
27 for (int outCh = 0; outCh < numOutputChannels; ++outCh) {
28 if (numInputChannels == numOutputChannels) {
29 // Direct copy
30 std::memcpy(output[outCh], input[outCh], sizeof(T) * numSamples);
31 } else if (numInputChannels > numOutputChannels) {
32 // Downmix: average groups of input channels
33 int groupSize = numInputChannels / numOutputChannels;
34 for (int n = 0; n < numSamples; ++n) {
35 T sum = 0;
36 for (int i = 0; i < groupSize; ++i)
37 sum += input[outCh * groupSize + i][n];
38 output[outCh][n] = sum / groupSize;
39 }
40 } else {
41 // Upmix: wrap or duplicate
42 int inCh = outCh % numInputChannels;
43 std::memcpy(output[outCh], input[inCh], sizeof(T) * numSamples);
44 }
45 }
46}

113template <typename T>
114void crossfadeBuffers(
115 const T* const* bufferA, const T* const* bufferB, T* const* output, size_t numChannels, size_t numSamples) {
116 for (size_t ch = 0; ch < numChannels; ++ch) {
117 for (size_t n = 0; n < numSamples; ++n) {
118 // Equal power crossfade
119 T fadeFactor = static_cast<T>(n) / static_cast<T>(numSamples - 1);
120 T gainA = std::cos(fadeFactor * (static_cast<T>(M_PI) / static_cast<T>(2))); // cos(0 to π/2)
121 T gainB = std::sin(fadeFactor * (static_cast<T>(M_PI) / static_cast<T>(2))); // sin(0 to π/2)
122 output[ch][n] = bufferA[ch][n] * gainA + bufferB[ch][n] * gainB;
123 }
124 }
125}
