59 void prepare(size_t numNewChannels, T newSampleRate) {
60 // Clamp and store number of channels and sample rate

64 // Prepare coefficient smoothers and gain state

85 T processSample(size_t ch, T targetGainDb) {
86 // What stage are we in?
87 T inAttack = static_cast<T>(targetGainDb < gainDb[ch]);
88 T inRelease = T(1) - inAttack;
89
90 // Compute the smoothing coefficient
91 T coeff =
92 inAttack * attackCoeff.getNextValue(ch) + inRelease * releaseCoeff.getNextValue(ch);
93 gainDb[ch] += coeff * (targetGainDb - gainDb[ch]);
94 return utils::dB2Mag(gainDb[ch]); // convert smoothed dB gain to linear
95 }

104 void processBlock(const T* const* input, T* const* output, size_t numSamples) {

171 void updateCoefficients(bool skipSmoothing = false) {
172 // Set target coefficients based on current attack and release times
173 attackCoeff.setTarget(1.0 - std::exp(-1.0 / (attackTimeSec * sampleRate)), skipSmoothing);
174 releaseCoeff.setTarget(1.0 - std::exp(-1.0 / (releaseTimeSec * sampleRate)), skipSmoothing);
