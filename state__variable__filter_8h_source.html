71 T processSample(size_t ch, T input) {
72 // Init output variables for multi-mode processing
73 T output, hp, bp, lp, ap;
74
75 // SERIES ROUTING
76 if constexpr (RoutingType == Routing::Series) {
77 output = input;
78 for (size_t s = 0; s < topology.getNumSections(); ++s) {
79 // Get current parameters for this channel and section
80 T g = design.getNextG(ch, s);
81 T twoR = design.getNextTwoR(ch, s);
82
83 // Process single sample through the topology for this channel and section.
84 topology.processSample(ch, s, output, g, twoR, hp, bp, lp);
85
86 // Select output based on the design's response mask for this channel and section.
87 const auto& responseMask = design.getResponseMask(ch, s);
88 output = hp * responseMask[static_cast<size_t>(Response::Highpass)] +
89 bp * responseMask[static_cast<size_t>(Response::Bandpass)] +
90 lp * responseMask[static_cast<size_t>(Response::Lowpass)];
91 }
92 }
93
94 // PARALLEL ROUTING
95 if constexpr (RoutingType == Routing::Parallel) {
96 for (size_t s = 0; s < topology.getNumSections(); ++s) {
97 // Get current parameters for this channel and section
98 T g = design.getNextG(ch, s);
99 T twoR = design.getNextTwoR(ch, s);
100
101 // Process single sample through the topology for this channel and section.
102 topology.processSample(ch, s, input, g, twoR, hp, bp, lp);
103
104 // Select output based on the design's response mask for this channel and section.
105 const auto& responseMask = design.getResponseMask(ch, s);
106 output += hp * responseMask[static_cast<size_t>(Response::Highpass)] +
107 bp * responseMask[static_cast<size_t>(Response::Bandpass)] +
108 lp * responseMask[static_cast<size_t>(Response::Lowpass)];
109 }
110 }
111 return output;
112 }

121 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
