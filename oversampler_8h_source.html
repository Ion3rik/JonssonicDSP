53 void prepare(size_t newNumChannels, size_t newMaxBlockSize) {
54 numChannels = newNumChannels;

56 // Prepare all stages
57 if constexpr (Factor >= 2) {
58 stage1.prepare(newNumChannels); // prepare stage 1
59 }
60 if constexpr (Factor >= 4) {
61 stage2.prepare(newNumChannels); // prepare stage 2
62 intermediateBuffer1to2.resize(newNumChannels,
63 newMaxBlockSize *

66 if constexpr (Factor >= 8) {

71 }

73 stage4.prepare(newNumChannels); // prepare stage 4
74 intermediateBuffer4to8.resize(newNumChannels,
75 newMaxBlockSize *

105 size_t upsample(const T* const* input, T* const* output, size_t numInputSamples) {
106 // Factor 1 (bypass)
107 if constexpr (Factor == 1) {
108 for (size_t ch = 0; ch < numChannels; ++ch) {
109 std::copy(input[ch], input[ch] + numInputSamples, output[ch]);
110 }
111 }
112 // Factor 2
113 if constexpr (Factor == 2) {
114 stage1.upsample(input, output, numInputSamples); // stage1 1x to 2x
115 }
116
117 // Factor 4
118 if constexpr (Factor == 4) {
119 // stage1 1x to 2x
120 stage1.upsample(input, intermediateBuffer1to2.writePtrs(), numInputSamples);
121 // stage2 2x to 4x
122 stage2.upsample(intermediateBuffer1to2.readPtrs(), output, 2 * numInputSamples);
123 }
124
125 // Factor 8
126 if constexpr (Factor == 8) {
127 // stage1 1x to 2x
128 stage1.upsample(input, intermediateBuffer1to2.writePtrs(), numInputSamples);
129 // stage2 2x to 4x
130 stage2.upsample(intermediateBuffer1to2.readPtrs(),
131 intermediateBuffer2to4.writePtrs(),
132 2 * numInputSamples);
133 // stage3 4x to 8x
134 stage3.upsample(intermediateBuffer2to4.readPtrs(), output, 4 * numInputSamples);
135 }
136
137 // Factor 16
138 if constexpr (Factor == 16) {
139 // stage1 1x to 2x
140 stage1.upsample(input, intermediateBuffer1to2.writePtrs(), numInputSamples);
141 // stage2 2x to 4x
142 stage2.upsample(intermediateBuffer1to2.readPtrs(),
143 intermediateBuffer2to4.writePtrs(),
144 2 * numInputSamples);
145 // stage3 4x to 8x
146 stage3.upsample(intermediateBuffer2to4.readPtrs(),
147 intermediateBuffer4to8.writePtrs(),
148 4 * numInputSamples);
149 // stage4 8x to 16x
150 stage4.upsample(intermediateBuffer4to8.readPtrs(), output, 8 * numInputSamples);
151 }
152 return numInputSamples * Factor;
153 }
161 void downsample(const T* const* input, T* const* output, size_t numOutputSamples) {
162 // Factor 1 (bypass)
163 if constexpr (Factor == 1) {
164 for (size_t ch = 0; ch < numChannels; ++ch) {
165 std::copy(input[ch], input[ch] + numOutputSamples, output[ch]);
166 }
167 }

178 stage1.downsample(intermediateBuffer1to2.readPtrs(), output, numOutputSamples);

185 // stage2 4x to 2x
186 stage2.downsample(intermediateBuffer2to4.readPtrs(),
187 intermediateBuffer1to2.writePtrs(),
188 2 * numOutputSamples);
189 // stage1 2x to 1x
190 stage1.downsample(intermediateBuffer1to2.readPtrs(), output, numOutputSamples);
191 }
192
193 // Factor 16

197 // stage3 8x to 4x
198 stage3.downsample(intermediateBuffer4to8.readPtrs(),
