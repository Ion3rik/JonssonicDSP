61 void prepare(size_t newNumChannels, T newSampleRate) {

108 T processSample(size_t ch, T input, T detectorInput) {
109 // FEEDFORWARD DETECTOR
110 if constexpr (DetectorType == DetectorType::Feedforward) {
111 // Side chain filtering if exists
112 if constexpr (SideChainFilter) {
113 detectorInput = sideChainFilter.processSample(ch, detectorInput);
114 }
115 // Route detector input to envelope follower
116 T env = envelopeFollower.processSample(ch, detectorInput);
117
118 // Gain computation
119 T gainDb = gainComputer.processSample(ch, env);
120
121 // Gain smoothing
122 T smoothGainLin = gainSmoother.processSample(ch, gainDb);
123
124 // Update max gain reduction for metering (max reduction is min gain in linear)
125 if constexpr (Metering)
126 maxGainReduction[ch] = std::min(maxGainReduction[ch], smoothGainLin);
127
128 // Apply gain
129 return input * smoothGainLin;
130 }
131 // FEEDBACK DETECTOR
132 else if constexpr (DetectorType == DetectorType::Feedback) {
133 // Detector signal from previous output
134 T detectorSignal = previousOutput[ch];
135
136 // Side chain filtering if exists
137 if constexpr (SideChainFilter) {
138 detectorSignal = sideChainFilter.processSample(ch, detectorSignal);
139 }
140
141 // Envelope follower
142 T env = envelopeFollower.processSample(ch, detectorSignal);
143
144 // Gain computation
145 T gainDb = gainComputer.processSample(ch, env);
146
147 // Gain smoothing
148 T smoothGainLin = gainSmoother.processSample(ch, gainDb);
149
150 // Update max gain reduction for metering (max reduction is min gain in linear)
151 if constexpr (Metering)
152 maxGainReduction[ch] = std::min(maxGainReduction[ch], smoothGainLin);
153
154 // Apply gain and update feedback state
155 previousOutput[ch] = input * smoothGainLin;
156 return previousOutput[ch];
157 }
158 }

170 void processBlock(const T* const* input,
171 const T* const* detectorInput,
172 T* const* output,
173 size_t numSamples,
174 T* gainReductionOutput = nullptr) {
175 // Reset metering state if enabled
176 if constexpr (Metering)
177 std::fill(maxGainReduction.begin(), maxGainReduction.end(), T(1));
178
179 // Process loop
180 for (size_t ch = 0; ch < numChannels; ++ch) {
181 for (size_t n = 0; n < numSamples; ++n) {
182 output[ch][n] = processSample(ch, input[ch][n], detectorInput[ch][n]);
183 }
184 // Output max gain reduction if enabled
185 if constexpr (Metering) {
186 if (gainReductionOutput) {
187 gainReductionOutput[ch] = maxGainReduction[ch];
188 }
189 }
190 }
191 }
