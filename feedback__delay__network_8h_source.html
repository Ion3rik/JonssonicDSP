65 void prepare(size_t newNumChannels, T newSampleRate, Time<T> newMaxDelay) {
66 // Clamp and store config variables
67 numChannels = utils::detail::clampChannels(newNumChannels);
68 sampleRate = utils::detail::clampSampleRate(newSampleRate);
69
70 // Prepare DSP components
71 Dm.prepare(M, sampleRate, newMaxDelay);
72 G.prepare(M, sampleRate);
73 modSource.prepare(M, sampleRate);

75 // Prepare mixing matrices and decay gains

113 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
114 for (size_t n = 0; n < numSamples; ++n) {
115 // Gather audio inputs for this sample
116 for (size_t ch = 0; ch < numChannels; ++ch)
117 inputFrame[ch] = input[ch][n];
118
119 // Input mixing: inputFrame (numChannels) -> x (M)
120 B.mix(inputFrame.data(), x.data());
121
122 // Read FDN delay line outputs with modulation and apply damping
123 for (size_t m = 0; m < M; ++m) {
124 T scaledMod = modDepthSamples.getNextValue(m) * Dm.getTargetDelay(m).toSamples(sampleRate) *
125 modSource.processSample(m);
126 s[m] = Dm.readSample(m, scaledMod);
127 s[m] = G.processSample(m, s[m]);
128 }
129
130 // Feedback mixing: s -> s
131 A.mix(s.data(), s.data());
132
133 // Write input + feedback to delay lines
134 for (size_t m = 0; m < M; ++m)
135 Dm.writeSample(m, x[m] + s[m]);
136
137 // Output mixing: s -> outputFrame (numChannels)
138 C.mix(s.data(), outputFrame.data());
139
140 // Write to audio outputs
141 for (size_t ch = 0; ch < numChannels; ++ch)
142 output[ch][n] = outputFrame[ch];
143 }
144 }

268 // Helper method
269 void updateDampingFilter() {
270 constexpr bool isShelf = std::is_same_v<DecayType, Shelf1Decay<T>> || std::is_same_v<DecayType, Shelf2Decay<T>>;
271 if constexpr (isShelf) {
272 for (size_t m = 0; m < M; ++m)
273 G.engine().setDecayTimes(m, Fc, RT60_LO, RT60_HI, Dm.getTargetDelay(m));
274 } else if constexpr (std::is_same_v<DecayType, OnePoleDecay<T>>) {
275 for (size_t m = 0; m < M; ++m)
276 G.engine().setDecayTimes(m, RT60_LO, RT60_HI, Dm.getTargetDelay(m));
277 }
