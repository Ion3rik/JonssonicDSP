64 void prepare(size_t newNumChannels, T newSampleRate, Time<T> newMaxDelay) {
65 // Clamp and store config variables
66 numChannels = utils::detail::clampChannels(newNumChannels);
67 sampleRate = utils::detail::clampSampleRate(newSampleRate);

112 void processBlock(const T* const* input, T* const* output, size_t numSamples) {
113 for (size_t n = 0; n < numSamples; ++n) {
114 // Gather audio inputs for this sample
115 for (size_t ch = 0; ch < numChannels; ++ch)
116 inputFrame[ch] = input[ch][n];
117
118 // Input mixing: inputFrame (numChannels) -> x (M)
119 B.mix(inputFrame.data(), x.data());
120
121 // Read FDN delay line outputs with modulation and apply damping
122 for (size_t m = 0; m < M; ++m) {
123 T scaledMod = modDepthSamples.getNextValue(m) * Dm.getTargetDelay(m).toSamples(sampleRate) *
124 modSource.processSample(m);
125 s[m] = Dm.readSample(m, scaledMod);
126 s[m] = G.processSample(m, s[m]);
127 }
128
129 // Feedback mixing: s -> s
130 A.mix(s.data(), s.data());
131
132 // Write input + feedback to delay lines
133 for (size_t m = 0; m < M; ++m)
134 Dm.writeSample(m, x[m] + s[m]);
135
136 // Output mixing: s -> outputFrame (numChannels)
137 C.mix(s.data(), outputFrame.data());
138
139 // Write to audio outputs
140 for (size_t ch = 0; ch < numChannels; ++ch)
141 output[ch][n] = outputFrame[ch];
142 }
143 }

267 // Helper method
268 void updateDampingFilter() {
269 if constexpr (DampingType == DampingType::FirstOrderShelf || DampingType == DampingType::BiquadShelf)
270 for (size_t m = 0; m < M; ++m)
271 G.setByT60(m, Fc, RT60_LO, RT60_HI, Dm.getTargetDelay(m));
272
273 if constexpr (DampingType == DampingType::OnePole)
274 for (size_t m = 0; m < M; ++m)
275 G.setByT60(m, RT60_LO, RT60_HI, Dm.getTargetDelay(m));
276 }
